PowerShell Scripts

 

Create-ADObjects

 

##Demo 02/24/2014

 

##Install Role

Add-WindowsFeature NET-Framework-Features -IncludeAllSubFeature

 

#Create New OU called Demo2

New-ADOrganizationalUnit `

    -Name Demo2 `

    -Path "dc=mbender,dc=Priv" `

    -ProtectedFromAccidentalDeletion $false -Verbose

 

 

#Create a new Global Security Group called Mbender-Admins-GS

New-ADGroup `

    -Name Mbender-Admins-GS `

    -GroupCategory Security `

    -GroupScope Global `

    -Path "ou=Demo2,dc=mbender,dc=priv" `

    -Verbose

 

 

#Create new User called mbender

New-ADUser `

    -Name mbender `

    -path "OU=Demo2,DC=mbender,DC=Priv" `

    -Enabled $true `

    -AccountPassword (ConvertTo-SecureString P@ssw0rd -AsPlainText  -force) `

    -Verbose

 

 

#Add mbender to Mbender-Admins-GS group

Add-ADGroupMember `

    -Identity Mbender-Admins-GS `

    -Members mbender `

    -Verbose

 

#add Mbender to Domain Admins

Add-ADGroupMember `

    -Identity "Domain Admins" `

    -Members mbender `

    -Verbose

#Region ... Create New AD User from existing object

New-ADUser `

    -Name mebender `

    -path "OU=Demo2,DC=mbender,DC=Priv" `

    -Enabled $true `

    -AccountPassword (ConvertTo-SecureString P@ssw0rd -AsPlainText  -force) `

    -Verbose -Instance mbender

#endregion

 

#create New GPO

New-GPO `

    -name WallpaperGPO-Admin `

    -Verbose

 

#Set GPO Settings

Set-GPRegistryValue `

    -Name WallpaperGPO-Admin `

     -key HKCU\Software\Microsoft\Windows\CurrentVersion\Policies\System `

     -ValueName Wallpaper `

     -Type String `

     -value \\ServerA\Data\admin.jpg `

     -Verbose

 

#Link GPO to OU

New-gplink `

    -name WallPaperGPO-Admin `

    -Target "OU=demo2,dc=mbender,dc=priv" `

    -LinkEnabled Yes `

    -Verbose

 

#Viewing AD Objects

Get-ADObject -Filter * -SearchBase "OU=demo2,dc=mbender,dc=priv"

 

##CleanUp

Remove-ADObject -Identity "ou=demo2,dc=mbender,dc=priv" -IncludeDeletedObjects -Recursive

Remove-GPO -Name WallPaperGPO-Admin
